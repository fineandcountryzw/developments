## PART 4: CRITICAL ISSUES IDENTIFIED

### 🔴 CRITICAL ISSUE #1: Commissions API IDOR Vulnerability
**Severity:** CRITICAL  
**Impact:** Any authenticated user can view any agent's commission data

**Evidence:**
```typescript
// app/api/agent/commissions/route.ts (lines 19-24)
export async function GET(request: NextRequest) {
  const { searchParams } = new URL(request.url);
  const agentId = searchParams.get('agentId'); // ❌ No auth check
  
  if (!agentId) {
    return apiError('agentId parameter is required');
  }
  // Direct DB query with user-supplied agentId - IDOR!
}
```

**Proof of Concept:**
```bash
# Agent A (user.id = 'agent-123') can view Agent B's commissions:
GET /api/agent/commissions?agentId=agent-456
# Returns Agent B's financial data without authorization check
```

**Fix Required:**
```typescript
export async function GET(request: NextRequest) {
  const authResult = await requireAgent();
  if (authResult.error) return authResult.error;
  const user = authResult.user;
  
  // Use authenticated user's ID, not query parameter
  const agentId = user.id;
  
  // Rest of logic...
}
```

**Files Affected:**
- `app/api/agent/commissions/route.ts` (main endpoint)
- `app/api/agent/commissions/analytics/route.ts` (needs verification)
- `app/api/agent/commissions/expected/route.ts` (needs verification)

**Typical Agent Workflow (Expected):**

2. Agent moves deal through stages (contacted → qualified → offer → closing)

**Current Workflow (Actual):**

- Add `PUT /api/agent/deals/[id]` (update stage, notes, closing date)
- Add `PATCH /api/agent/deals/[id]/close` (mark as won/lost)

```typescript
// app/api/agent/deals/route.ts

export async function POST(request: NextRequest) {
  try {
    const authResult = await requireAgent();
    if (authResult.error) return authResult.error;
    const user = authResult.user;
    
    const body = await request.json();
    const { title, clientId, stageId, value, expectedCloseDate, probability, notes } = body;
    
    // Validation
    if (!title || !clientId || !stageId) {
      return apiError('Title, client, and stage are required', 400, ErrorCodes.VALIDATION_ERROR);
    }
    
    // Verify client belongs to agent
    const client = await prisma.client.findUnique({
      where: { id: clientId },
      include: { reservations: { where: { agentId: user.id } } }
    });
    
    if (!client || client.reservations.length === 0) {
      return apiError('Unauthorized - not your client', 403, ErrorCodes.ACCESS_DENIED);
    }
    
    // Create deal
    const deal = await prisma.deal.create({
      data: {
        title,
        clientId,
        stageId,
        ownerId: user.id,
        value: value || 0,
        expectedCloseDate: expectedCloseDate ? new Date(expectedCloseDate) : null,
        probability: probability || 50,
        notes: notes || null,
        isArchived: false
      }
    });
    
    return apiSuccess({ deal, message: 'Deal created successfully' }, 201);
    
  } catch (error: any) {
    logger.error('Error creating deal', error, { module: 'API', action: 'POST_AGENT_DEALS' });
    return apiError(error.message || 'Failed to create deal', 500, ErrorCodes.CREATE_ERROR);
  }
}
```

```typescript
// app/api/agent/deals/[id]/route.ts

export async function PUT(request: NextRequest, { params }: { params: Promise<{ id: string }> }) {
  try {
    const authResult = await requireAgent();
    if (authResult.error) return authResult.error;
    const user = authResult.user;
    
    const { id } = await params;
    const body = await request.json();
    const { stageId, value, expectedCloseDate, probability, notes } = body;
    
    // Get existing deal
    const existingDeal = await prisma.deal.findUnique({
      where: { id }
    });
    
    if (!existingDeal) {
      return apiError('Deal not found', 404, ErrorCodes.NOT_FOUND);
    }
    
    // Verify ownership
    if (existingDeal.ownerId !== user.id) {
      return apiError('Unauthorized - not your deal', 403, ErrorCodes.ACCESS_DENIED);
    }
    
    // Update deal
    const updatedDeal = await prisma.deal.update({
      where: { id },
      data: {
        ...(stageId && { stageId }),
        ...(value !== undefined && { value }),
        ...(expectedCloseDate && { expectedCloseDate: new Date(expectedCloseDate) }),
        ...(probability !== undefined && { probability }),
        ...(notes !== undefined && { notes })
      }
    });
    
    return apiSuccess({ deal: updatedDeal, message: 'Deal updated successfully' });
    
  } catch (error: any) {
    logger.error('Error updating deal', error, { module: 'API', action: 'PUT_AGENT_DEAL' });
    return apiError(error.message || 'Failed to update deal', 500, ErrorCodes.UPDATE_ERROR);
  }
}
```
